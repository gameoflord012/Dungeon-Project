tree "Root"
    fallback
        while GetChaseTarget
            fallback
                tree "Attack"
                ChaseTarget
                        
        while TargetEscaped
            Sequence
                StopMoving
                Wait 1.0
                tree "LookAround"
                ResetTarget
                
        while not GetChaseTarget
            tree "Patrol"
        
tree "Patrol"
    sequence
        AdvancedPath
        MoveToPatrolPosition
        Wait .5
        tree "LookAround"
        

tree "LookAround"
    Sequence        
        LookAround(90)
        Wait 2.0
        LookAround(-180)
        Wait 2.0

tree "Attack"   
    Sequence
        IsInAttackRange
        AttackTarget
        Wait .5
